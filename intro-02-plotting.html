using Plots

# ∇ = \nabla <TAB>
function ∇f(x, y; scale=hypot(x, y)^0.5 * 3)
    return [-y, x] ./ scale
end

# x mesh and y mesh
r = -1.0:0.2:1.0
xx = [x for y in r, x in r]
yy = [y for y in r, x in r]

quiver(xx, yy, quiver=∇f, aspect_ratio=:equal, line=(:black))
