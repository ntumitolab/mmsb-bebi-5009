Fig 6.18#

Model of calcium-induced calcium release in hepatocytes

using Catalyst
using ModelingToolkit
using OrdinaryDiffEq
using Plots
Plots.default(linewidth=2)

rn = @reaction_network begin
    @parameters I(t)
    (k1 * I, km1), R <--> RI
    (k2 * C, km2), RI <--> RIC
    (k3 * C, km3), RIC <--> RICC
    vr * (γ0 + γ1 * RIC) * (Cer - C), 0 --> C
    hill(C, p1, p2, 4), C => 0
end
\[\begin{split} \begin{align*} \mathrm{R} &\xrightleftharpoons[\mathtt{km1}]{\mathtt{k1} I\left( t \right)} \mathrm{\mathtt{RI}} \\ \mathrm{\mathtt{RI}} &\xrightleftharpoons[\mathtt{km2}]{C \mathtt{k2}} \mathrm{\mathtt{RIC}} \\ \mathrm{\mathtt{RIC}} &\xrightleftharpoons[\mathtt{km3}]{C \mathtt{k3}} \mathrm{\mathtt{RICC}} \\ \varnothing &\xrightleftharpoons[\frac{C^{4} \mathtt{p1}}{C^{4} + \mathtt{p2}^{4}}]{\left( - C + \mathtt{Cer} \right) \left( \mathtt{{\gamma}0} + \mathtt{RIC} \mathtt{{\gamma}1} \right) \mathtt{vr}} \mathrm{C} \end{align*} \end{split}\]
setdefaults!(rn, [
    :γ0 => 0.1,
    :γ1 => 20.5,
    :p1 => 8.5,
    :p2 => 0.065,
    :k1 => 12,
    :k2 => 15,
    :k3 => 1.8,
    :km1 => 8,
    :km2 => 1.65,
    :km3 => 0.21,
    :Cer => 8.37,
    :vr => 0.185,
    :I => 0,
    :C => 0,
    :R => 1,
    :RI => 0,
    :RIC => 0,
    :RICC => 0
])

osys = convert(ODESystem, rn; remove_conserved = true) |> structural_simplify
equations(osys)

discrete_events = [[20] => [I ~ 0.7], [60] => [I ~ 1.2], [90] => [I ~ 4.0]]
